## 一、用户系统

```sql
CREATE TABLE `user` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `username` VARCHAR(32) NOT NULL UNIQUE,
  `nickname` VARCHAR(64),
  `avatar` VARCHAR(512),
  `email` VARCHAR(128),
  `phone` VARCHAR(20),
  `password_hash` VARCHAR(128),
  `role` ENUM('student','teacher','admin') DEFAULT 'student',
  `level` INT DEFAULT 1,
  `exp` BIGINT DEFAULT 0,
  `points` BIGINT DEFAULT 0,
  `title` VARCHAR(64),
  `bio` VARCHAR(500),
  `status` TINYINT DEFAULT 1,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_level` (`level`),
  INDEX `idx_points` (`points`)
);

CREATE TABLE `user_follow` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `follower_id` BIGINT NOT NULL,
  `following_id` BIGINT NOT NULL,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_follow` (`follower_id`, `following_id`)
);
```

## 二、班级与宠物

```sql
CREATE TABLE `class` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `name` VARCHAR(64) NOT NULL,
  `teacher_id` BIGINT NOT NULL,
  `grade` VARCHAR(32),
  `school_year` VARCHAR(16),
  `pet_name` VARCHAR(64) DEFAULT '班级萌兽',
  `pet_level` INT DEFAULT 1,
  `pet_exp` BIGINT DEFAULT 0,
  `total_points` BIGINT DEFAULT 0,
  `student_count` INT DEFAULT 0,
  `status` TINYINT DEFAULT 1,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_teacher` (`teacher_id`)
);

CREATE TABLE `class_student` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `class_id` BIGINT NOT NULL,
  `user_id` BIGINT NOT NULL,
  `joined_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_class_user` (`class_id`, `user_id`)

-- 班级宠物等级配置
CREATE TABLE `pet_level_config` (
  `level` INT PRIMARY KEY,
  `name` VARCHAR(32) NOT NULL,
  `required_exp` BIGINT NOT NULL,
  `avatar_url` VARCHAR(512),
  `description` VARCHAR(256)
);
```

## 三、创作系统

```sql
CREATE TABLE `project` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `title` VARCHAR(128) NOT NULL,
  `description` TEXT,
  `sb3_url` VARCHAR(512),
  `cover_url` VARCHAR(512),
  `visibility` ENUM('public','private','class') DEFAULT 'public',
  `tags` JSON,
  `like_count` INT DEFAULT 0,
  `collect_count` INT DEFAULT 0,
  `view_count` INT DEFAULT 0,
  `comment_count` INT DEFAULT 0,
  `is_featured` TINYINT DEFAULT 0,
  `status` ENUM('draft','published','reviewing','rejected') DEFAULT 'draft',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX `idx_user` (`user_id`),
  INDEX `idx_likes` (`like_count` DESC),
  FULLTEXT `idx_search` (`title`, `description`)
);

CREATE TABLE `project_like` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `project_id` BIGINT NOT NULL,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_user_project` (`user_id`, `project_id`)
);

CREATE TABLE `project_collect` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `project_id` BIGINT NOT NULL,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_user_project` (`user_id`, `project_id`)
);

CREATE TABLE `project_comment` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `project_id` BIGINT NOT NULL,
  `user_id` BIGINT NOT NULL,
  `parent_id` BIGINT,
  `content` TEXT NOT NULL,
  `status` TINYINT DEFAULT 1,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_project` (`project_id`)
);
```

## 四、sb3 解析结果

```sql
-- sb3 解析结果主表
CREATE TABLE `sb3_parse_result` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `project_id` BIGINT,              -- 关联作品(可为空,作业/OJ场景独立)
  `user_id` BIGINT NOT NULL,
  `sb3_url` VARCHAR(512) NOT NULL,
  `parse_status` ENUM('pending','success','failed') DEFAULT 'pending',
  `scratch_version` VARCHAR(16),
  `canvas_width` INT,
  `canvas_height` INT,
  `sprite_count` INT DEFAULT 0,
  `script_count` INT DEFAULT 0,
  `total_blocks` INT DEFAULT 0,
  `variable_count` INT DEFAULT 0,
  `list_count` INT DEFAULT 0,
  `broadcast_count` INT DEFAULT 0,
  `costume_count` INT DEFAULT 0,
  `sound_count` INT DEFAULT 0,
  `extension_count` INT DEFAULT 0,
  `max_nesting_depth` INT DEFAULT 0,
  `complexity_score` DECIMAL(5,2) DEFAULT 0,
  `block_type_stats` JSON,          -- {"event":3,"control":8,"motion":12,...}
  `full_parse_data` JSON,           -- 完整解析结果(角色/程序链/素材等)
  `ai_comment` TEXT,                -- AI 点评文本
  `ai_score` DECIMAL(5,2),          -- AI 评分
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_user` (`user_id`),
  INDEX `idx_project` (`project_id`)
);
```

## 五、题库系统

```sql
CREATE TABLE `problem` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `title` VARCHAR(256) NOT NULL,
  `description` TEXT NOT NULL,
  `problem_type` ENUM('choice','fill','scratch_algo','scratch_creative') NOT NULL,
  `difficulty` ENUM('easy','medium','hard','expert') NOT NULL,
  `tags` JSON,
  `score` INT DEFAULT 100,
  `test_cases` JSON,                -- 算法题测试用例
  `sample_project_url` VARCHAR(512), -- 创意题参考作品
  `ai_solution` TEXT,
  `ai_hint` TEXT,
  `ac_count` INT DEFAULT 0,
  `submit_count` INT DEFAULT 0,
  `author_id` BIGINT,
  `status` ENUM('draft','published','disabled') DEFAULT 'draft',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_difficulty` (`difficulty`)
);

CREATE TABLE `problem_set` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `title` VARCHAR(128) NOT NULL,
  `description` TEXT,
  `problem_ids` JSON,
  `difficulty` ENUM('easy','medium','hard'),
  `category` VARCHAR(64),
  `created_by` BIGINT,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE `submission` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `problem_id` BIGINT NOT NULL,
  `sb3_url` VARCHAR(512),
  `sb3_parse_id` BIGINT,            -- 关联 sb3 解析结果
  `result` ENUM('pending','judging','AC','WA','TLE','MLE','RE','CE') DEFAULT 'pending',
  `score` INT DEFAULT 0,
  `test_results` JSON,
  `ai_comment` TEXT,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_user_problem` (`user_id`, `problem_id`),
  INDEX `idx_result` (`result`)
);
```

## 六、作业系统

```sql
CREATE TABLE `homework` (
```

## 十三、ER 核心链路

```
user ──N:N──> class (via class_student)
  │              │
  │              └── 班级宠物积分 (class.pet_level/total_points)
  │
  ├──1:N──> project ──1:1──> sb3_parse_result
  │              │                │
  │              │                └── full_parse_data (MongoDB)
  │              ├──1:N──> project_comment
  │              ├──N:N──> project_like
  │              └──N:N──> project_collect
  │
  ├──1:N──> submission ──N:1──> problem
  │                              │
  │                              └── test_cases
  │
  ├──1:N──> homework_submission ──N:1──> homework ──N:1──> class
  │
  ├──1:N──> points_log
  ├──1:N──> notification
  ├──N:N──> badge (via user_badge)
  └──1:N──> learning_progress ──N:1──> course ──1:N──> course_unit
```
