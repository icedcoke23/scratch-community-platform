## 一、用户系统（3 张）

```sql
-- ============================================================
-- 1. user — 用户主表
-- 包含竞赛 Rating 预留字段（Phase 3 启用）
-- ============================================================
CREATE TABLE `user` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `username` VARCHAR(32) NOT NULL UNIQUE,
  `nickname` VARCHAR(64),
  `avatar` VARCHAR(512),
  `email` VARCHAR(128),
  `phone` VARCHAR(20),
  `password_hash` VARCHAR(128),
  `role` ENUM('student','teacher','admin') DEFAULT 'student',
  `level` INT DEFAULT 1,
  `exp` BIGINT DEFAULT 0,
  `points` BIGINT DEFAULT 0,
  `title` VARCHAR(64),
  `bio` VARCHAR(500),
  `status` TINYINT DEFAULT 1,
  -- 竞赛 Rating 预留（Phase 3 启用）
  `contest_rating` INT DEFAULT 800 COMMENT '竞赛积分，初始 800，Phase 3 启用',
  `contest_color` VARCHAR(20) DEFAULT 'newbie' COMMENT '颜色标识：newbie/learner/practitioner/proficient/master',
  `contest_count` INT DEFAULT 0 COMMENT '参赛场次',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_level` (`level`),
  INDEX `idx_points` (`points`),
  INDEX `idx_contest_rating` (`contest_rating`)
) COMMENT '用户主表';

-- ============================================================
-- 2. user_follow — 用户关注关系
-- ============================================================
CREATE TABLE `user_follow` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `follower_id` BIGINT NOT NULL,
  `following_id` BIGINT NOT NULL,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_follow` (`follower_id`, `following_id`),
  INDEX `idx_following` (`following_id`)
) COMMENT '用户关注关系';

-- ============================================================
-- 3. class — 班级
-- ============================================================
CREATE TABLE `class` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `name` VARCHAR(64) NOT NULL,
  `teacher_id` BIGINT NOT NULL,
  `grade` VARCHAR(32),
  `school_year` VARCHAR(16),
  `invite_code` VARCHAR(16) UNIQUE COMMENT '班级邀请码',
  `pet_name` VARCHAR(64) DEFAULT '班级萌兽',
  `pet_level` INT DEFAULT 1,
  `pet_exp` BIGINT DEFAULT 0,
  `total_points` BIGINT DEFAULT 0,
  `student_count` INT DEFAULT 0,
  `status` TINYINT DEFAULT 1,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_teacher` (`teacher_id`)
) COMMENT '班级';

-- ============================================================
-- 4. class_student — 班级学生关联
-- ============================================================
CREATE TABLE `class_student` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `class_id` BIGINT NOT NULL,
  `user_id` BIGINT NOT NULL,
  `joined_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_class_user` (`class_id`, `user_id`),
  INDEX `idx_user` (`user_id`)
) COMMENT '班级-学生关联表';
```

## 二、创作系统（4 张）

```sql
-- ============================================================
-- 5. project — 项目/作品主表
-- 包含 Remix 归属链字段（root_id/parent_id/remix_count）
-- ============================================================
CREATE TABLE `project` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `title` VARCHAR(128) NOT NULL,
  `description` TEXT,
  `sb3_url` VARCHAR(512),
  `cover_url` VARCHAR(512),
  `visibility` ENUM('public','private','class') DEFAULT 'public',
  `tags` JSON,
  `like_count` INT DEFAULT 0,
  `view_count` INT DEFAULT 0,
  `comment_count` INT DEFAULT 0,
  `is_featured` TINYINT DEFAULT 0,
  `status` ENUM('draft','published','reviewing','rejected') DEFAULT 'draft',
  -- Remix 归属链（Phase 1 即启用）
  `root_id` BIGINT COMMENT '最原始项目 ID，穿透所有 remix 层级',
  `parent_id` BIGINT COMMENT '直接父项目 ID（被 remix 的项目）',
  `remix_count` INT DEFAULT 0 COMMENT '被 remix 次数',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX `idx_user` (`user_id`),
  INDEX `idx_likes` (`like_count` DESC),
  INDEX `idx_root` (`root_id`),
  INDEX `idx_parent` (`parent_id`),
  INDEX `idx_featured` (`is_featured`, `created_at` DESC),
  FULLTEXT `idx_search` (`title`, `description`)
) COMMENT '项目/作品主表';

-- ============================================================
-- 6. project_like — 项目点赞
-- ============================================================
CREATE TABLE `project_like` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `project_id` BIGINT NOT NULL,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_user_project` (`user_id`, `project_id`),
  INDEX `idx_project` (`project_id`)
) COMMENT '项目点赞';

-- ============================================================
-- 7. project_comment — 项目评论
-- ============================================================
CREATE TABLE `project_comment` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `project_id` BIGINT NOT NULL,
  `user_id` BIGINT NOT NULL,
  `parent_id` BIGINT COMMENT '回复的父评论 ID，NULL 为顶层评论',
  `content` TEXT NOT NULL,
  `status` TINYINT DEFAULT 1 COMMENT '1=正常 0=删除',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_project` (`project_id`),
  INDEX `idx_parent` (`parent_id`)
) COMMENT '项目评论';

-- ============================================================
-- 8. sb3_parse_result — SB3 解析结果
-- 存储项目结构化解析数据，供判题和展示使用
-- ============================================================
CREATE TABLE `sb3_parse_result` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `project_id` BIGINT COMMENT '关联作品，可为空（作业/OJ 场景独立）',
  `user_id` BIGINT NOT NULL,
  `sb3_url` VARCHAR(512) NOT NULL,
  `parse_status` ENUM('pending','success','failed') DEFAULT 'pending',
  `scratch_version` VARCHAR(16),
  `canvas_width` INT,
  `canvas_height` INT,
  `sprite_count` INT DEFAULT 0,
  `script_count` INT DEFAULT 0,
  `total_blocks` INT DEFAULT 0,
  `variable_count` INT DEFAULT 0,
  `list_count` INT DEFAULT 0,
  `broadcast_count` INT DEFAULT 0,
  `costume_count` INT DEFAULT 0,
  `sound_count` INT DEFAULT 0,
  `extension_count` INT DEFAULT 0,
  `max_nesting_depth` INT DEFAULT 0,
  `complexity_score` DECIMAL(5,2) DEFAULT 0,
  `block_type_stats` JSON COMMENT '积木类型统计 {"event":3,"control":8,...}',
  `full_parse_data` JSON COMMENT '完整解析结果（角色/程序链/素材等）',
  `ai_comment` TEXT COMMENT 'AI 点评文本',
  `ai_score` DECIMAL(5,2) COMMENT 'AI 评分',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_user` (`user_id`),
  INDEX `idx_project` (`project_id`),
  INDEX `idx_status` (`parse_status`)
) COMMENT 'SB3 文件解析结果';
```

## 三、题库 + 判题（3 张表，含作业 2 张共 5 个逻辑实体）

```sql
-- ============================================================
-- 9. problem — 题目主表
-- problem_type 包含 scratch_effect 等 v0.4 定义的类型
-- ============================================================
CREATE TABLE `problem` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `title` VARCHAR(256) NOT NULL,
  `description` TEXT NOT NULL,
  `problem_type` ENUM(
    'choice',              -- 选择题
    'fill',                -- 填空题
    'scratch_algo',        -- Scratch 算法题（判题）
    'scratch_creative',    -- Scratch 创意题（AI 评分）
    'scratch_effect'       -- Scratch 效果题（视觉效果验证）
  ) NOT NULL,
  `difficulty` ENUM('easy','medium','hard','expert') NOT NULL,
  `tags` JSON COMMENT '标签数组 ["循环","变量","事件"]',
  `score` INT DEFAULT 100,
  `test_cases` JSON COMMENT '算法题测试用例',
  `sample_project_url` VARCHAR(512) COMMENT '创意题参考作品',
  `ai_solution` TEXT COMMENT 'AI 标准解法',
  `ai_hint` TEXT COMMENT 'AI 提示',
  `ac_count` INT DEFAULT 0,
  `submit_count` INT DEFAULT 0,
  `author_id` BIGINT,
  `status` ENUM('draft','published','disabled') DEFAULT 'draft',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX `idx_difficulty` (`difficulty`),
  INDEX `idx_type` (`problem_type`),
  INDEX `idx_status` (`status`),
  FULLTEXT `idx_search` (`title`, `description`)
) COMMENT '题目主表';

-- ============================================================
-- 10. submission — 提交记录
-- ============================================================
CREATE TABLE `submission` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `problem_id` BIGINT NOT NULL,
  `sb3_url` VARCHAR(512),
  `sb3_parse_id` BIGINT COMMENT '关联 sb3_parse_result',
  `result` ENUM('pending','judging','AC','WA','TLE','MLE','RE','CE') DEFAULT 'pending',
  `score` INT DEFAULT 0,
  `test_results` JSON COMMENT '测试用例执行结果',
  `ai_comment` TEXT COMMENT 'AI 评语',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_user_problem` (`user_id`, `problem_id`),
  INDEX `idx_result` (`result`),
  INDEX `idx_problem` (`problem_id`),
  INDEX `idx_time` (`created_at`)
) COMMENT '判题提交记录';

-- ============================================================
-- 11. homework — 作业
-- ============================================================
CREATE TABLE `homework` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `teacher_id` BIGINT NOT NULL,
  `class_id` BIGINT NOT NULL,
  `title` VARCHAR(128) NOT NULL,
  `description` TEXT,
  `homework_type` ENUM('free_create','from_problem','from_course_unit') NOT NULL COMMENT '自由创作/指定题目/课程单元',
  `problem_id` BIGINT COMMENT '指定题目 ID（homework_type=from_problem 时）',
  `course_unit_id` BIGINT COMMENT '课程单元 ID（homework_type=from_course_unit 时）',
  `deadline` DATETIME,
  `total_score` INT DEFAULT 100,
  `auto_grade` TINYINT DEFAULT 1 COMMENT '是否自动评分',
  `status` ENUM('draft','published','closed') DEFAULT 'draft',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_class` (`class_id`),
  INDEX `idx_teacher` (`teacher_id`),
  INDEX `idx_deadline` (`deadline`)
) COMMENT '作业';

-- ============================================================
-- 12. homework_submission — 作业提交
-- ============================================================
CREATE TABLE `homework_submission` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `homework_id` BIGINT NOT NULL,
  `user_id` BIGINT NOT NULL,
  `sb3_url` VARCHAR(512),
  `sb3_parse_id` BIGINT COMMENT '关联 sb3_parse_result',
  `auto_score` INT COMMENT '自动评分',
  `teacher_score` INT COMMENT '教师评分',
  `final_score` INT COMMENT '最终评分',
  `teacher_comment` TEXT COMMENT '教师评语',
  `ai_comment` TEXT COMMENT 'AI 评语',
  `status` ENUM('pending','submitted','auto_graded','teacher_graded') DEFAULT 'pending',
  `submitted_at` DATETIME,
  `graded_at` DATETIME,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_homework_user` (`homework_id`, `user_id`),
  INDEX `idx_user` (`user_id`),
  INDEX `idx_status` (`status`)
) COMMENT '作业提交记录';
```

## 四、积分 + 通知 + 审核（3 张）

```sql
-- ============================================================
-- 13. points_log — 积分流水
-- ============================================================
CREATE TABLE `points_log` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,

  `amount` INT NOT NULL COMMENT '变动金额，正数为获得，负数为消耗',

-- ============================================================
-- 14. notification — 站内通知
-- ============================================================
CREATE TABLE `notification` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `type` ENUM('homework','grade','social','points','system','reminder') NOT NULL,
  `title` VARCHAR(128) NOT NULL,
  `content` VARCHAR(512),
  `data` JSON COMMENT '附加数据，如链接/跳转参数',
  `is_read` TINYINT DEFAULT 0,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_user_read` (`user_id`, `is_read`),
  INDEX `idx_time` (`created_at`)
) COMMENT '站内通知';

-- ============================================================
-- 15. content_audit_log — 内容审核日志
-- ============================================================
CREATE TABLE `content_audit_log` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `content_type` ENUM('project','comment','asset') NOT NULL,
  `content_id` BIGINT NOT NULL,
  `user_id` BIGINT NOT NULL COMMENT '内容作者',
  `audit_stage` ENUM('rule','ai','manual') NOT NULL,
  `audit_result` ENUM('pass','reject','pending') NOT NULL,
  `confidence` DECIMAL(3,2) COMMENT 'AI 审核置信度',
  `reason` VARCHAR(500),
  `auditor_id` BIGINT COMMENT '人工审核员 ID',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_content` (`content_type`, `content_id`),
  INDEX `idx_result` (`audit_result`),
  INDEX `idx_user` (`user_id`)
) COMMENT '内容审核日志';
```

## 五、课程系统（2 张 + learning_progress 合并入 course_unit）

```sql
-- ============================================================
-- 16. course — 课程
-- ============================================================
CREATE TABLE `course` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `title` VARCHAR(128) NOT NULL,
  `description` TEXT,
  `cover_url` VARCHAR(512),
  `difficulty` ENUM('beginner','intermediate','advanced') NOT NULL,
  `unit_count` INT DEFAULT 0,
  `student_count` INT DEFAULT 0,
  `is_shared` TINYINT DEFAULT 0 COMMENT '是否共享给其他教师',
  `created_by` BIGINT NOT NULL,
  `status` ENUM('draft','published','disabled') DEFAULT 'draft',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX `idx_creator` (`created_by`),
  INDEX `idx_difficulty` (`difficulty`)
) COMMENT '课程';

-- ============================================================
-- 17. course_unit — 课程单元 + 学习进度
-- learning_progress 合并为 progress_* 字段，按 (user_id, unit_id) 唯一
-- ============================================================
CREATE TABLE `course_unit` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `course_id` BIGINT NOT NULL,
  `title` VARCHAR(128) NOT NULL,
  `intro` VARCHAR(512) COMMENT '单元简介',
  `cover_url` VARCHAR(512),
  `video_url` VARCHAR(512),
  `content` LONGTEXT COMMENT '单元正文内容（Markdown/HTML）',
  `example_project_url` VARCHAR(512) COMMENT '示例作品 URL',
  `problem_id` BIGINT COMMENT '关联练习题',
  `order_num` INT DEFAULT 0 COMMENT '排序序号',
  `map_x` INT COMMENT '地图 X 坐标（关卡地图展示）',
  `map_y` INT COMMENT '地图 Y 坐标',
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX `idx_course` (`course_id`),
  INDEX `idx_order` (`course_id`, `order_num`)
) COMMENT '课程单元';

-- ============================================================
-- 18. learning_progress — 学习进度（独立表，按 user+unit 粒度）
-- 从 course_unit 分离，支持按单元追踪进度
-- ============================================================
CREATE TABLE `learning_progress` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `user_id` BIGINT NOT NULL,
  `course_id` BIGINT NOT NULL,
  `unit_id` BIGINT NOT NULL,
  `status` ENUM('not_started','in_progress','completed') DEFAULT 'not_started',
  `progress` DECIMAL(5,2) DEFAULT 0 COMMENT '完成百分比 0.00-100.00',
  `score` INT COMMENT '单元测验得分',
  `started_at` DATETIME,
  `completed_at` DATETIME,
  `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  UNIQUE KEY `uk_user_unit` (`user_id`, `unit_id`),
  INDEX `idx_course` (`course_id`),
  INDEX `idx_user_course` (`user_id`, `course_id`)
) COMMENT '学习进度（按单元粒度）';
```

## 🔗 ER 核心链路

```
user ──N:N──> class (via class_student)
  │
  ├──1:N──> project ──1:1──> sb3_parse_result
  │              │
  │              ├──1:N──> project_comment (支持楼中楼 parent_id)
  │              ├──N:N──> project_like (via project_like)
  │              └── Remix: root_id → 原始项目, parent_id → 父项目
  │
  ├──1:N──> submission ──N:1──> problem
  │
  ├──1:N──> homework (N:1 → class)
  │              └──1:N──> homework_submission
  │
  ├──1:N──> points_log
  ├──1:N──> notification
  ├──1:N──> content_audit_log
  │
  └── learning_progress ──N:1──> course_unit ──N:1──> course
```
